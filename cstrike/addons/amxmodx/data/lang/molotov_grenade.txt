[en]
CVAR_ACCESS = Access flags for purchase or equip. Empty string = everyone; "abc" = require matching flags
CVAR_EQUIP_ACCESS = Auto-equip grenade at round start (0 = off, 1 = on)
CVAR_CHECK_BUYZONE = Require player to be in the buy zone (0 = off, 1 = on)
CVAR_COST = Grenade price
CVAR_BUY_LIMIT = Purchase limit per round (-1 = unlimited)
CVAR_LIMIT_ROUND = Grenade unlocks only after this round number (0 = from round 1)
CVAR_HIT_PLAYER = Damage on direct grenade hit
CVAR_KILLFEED = Show [ᴍᴏʟᴏᴛᴏᴠ] / [ɪɴᴄ] prefix next to the name in killfeed (0 = off, 1 = on)
CVAR_RADIUS = Fire zone radius in units
CVAR_THROWTIME = Flight time before auto-detonation (seconds)
CVAR_DURATION = Burn duration (seconds)
CVAR_DAMAGE_MODE = Who takes damage: 0 = enemies only, 1 = enemies and thrower, 2 = all players
CVAR_DAMAGE_RADIUS_MODE = Damage method: 1 = fixed anywhere in the zone (Ham_TakeDamage), 2 = scales with distance and armor (rg_dmg_radius)
CVAR_DAMAGE_TIME = Damage tick interval (seconds)
CVAR_DAMAGE_VALUE = Base damage per tick
CVAR_DAMAGE_MAX_MULT = Maximum damage multiplier for prolonged fire exposure (1.0 = no scaling, 3.0 = triple damage at peak)
CVAR_EFFECT_NUM = Number of debris sprites (floor fire particles)
CVAR_SMOKE_TOUCH = Extinguish with smoke grenade (0 = off, 1 = on)
CVAR_SMOKE_OWNER = Path to smoke grenade model (default: models/w_smokegrenade.mdl)
CVAR_SMOKE_DURATION = Smoke suppression zone lifetime (seconds)
CVAR_WATER_TOUCH = Extinguish on water contact (0 = off, 1 = on)
CVAR_SKY_FORCE = Remove grenade on skybox contact (0 = off, 1 = on)
CVAR_SPREAD_ENABLE = Fire spread mode: burning radius expands at half of burn duration (0 = off, 1 = on)
CVAR_SPREAD_RADIUS = Units added to burning radius when spread triggers
NTF_BUY_LIMIT = You have reached the purchase limit for this round
NTF_VIP = You are VIP. Purchase not available
NTF_WPN_NA = Grenade is not available to you
NTF_WPN_CANTBUY = You cannot buy the grenade
NTF_WPN_CANTBUY_TIME = Buy time has expired
NTF_ALREADY_HO = You already have a grenade
NTF_NOT_MONEY = Not enough money
NTF_CARRY_ANY = You cannot carry more grenades
